00024 /**
00025 * Template based ValueUpdater implementation.
00026 *
00027 * This updater functions on non-const methods which take Objects, for an
00028 * updater which works with non-Object types see BoxUpdater.
00029 *
00030 * For ease of use the COH_TYPED_UPDATER macro can be used to easily construct
00031 * an instance of this class. For example the following constructs an updater
00032 * for calling the "void Address::setState(String::View)" method.
00033 *
00034 * @code
00035 * ValueUpdater::View vUpd = COH_TYPED_UPDATER(Address, setState, String::View);
00036 * @endcode
00037 *
00038 * @author mf 2009.07.29
00039 *
00040 * @see BoxUpdater
00041 */
00042 template<class A, class C, void (C::*M)(A), class AH = A, class OH = typename C::Handle>
00043 class TypedUpdater
00044     : public class_spec<TypedUpdater<A, C, M, AH, OH>,
00045         extends<ReflectionUpdater> >
00046     {
00047     friend class factory<TypedUpdater<A, C, M, AH, OH> >;
00048 
00049     // ----- constructors ---------------------------------------------------
00050 
00051     protected:
00052         /**
00053         * Construct a TypedUpdater
00054         */
00055         TypedUpdater()
00056             {
00057             }
00058 
00059         /**
00060         * Construct a TypedUpdater based on a method name and optional
00061         * parameters.
00062         *
00063         * The method name is only used for the purposes of serializing the
00064         * updater for execution on remote Java members.
00065         *
00066         * @param vsMethod  the name of the method to invoke via reflection
00067         */
00068         TypedUpdater(String::View vsMethod)
00069             : class_spec<TypedUpdater<A, C, M, AH, OH>,
00070             extends<ReflectionUpdater> >(vsMethod)
00071             {
00072             }
00073 
00074 
00075    // ----- ValueUpdater interface ---------------------------------------
00076 
00077     public:
00078         /**
00079         * {@inheritDoc}
00080         */
00081         virtual void update(Object::Handle hTarget, Object::Holder ohValue) const
00082             {
00083             // Bug19826856,COH-12476; strict aliasing errors
00084             // casting to local variable seems to be necessary, suspect
00085             // a compiler issue, GCC 4.8.2
00086             // get_pointer also needed for GCC 4.8.2
00087             OH hOHTarget = cast<OH>(hTarget);
00088             C* pTarget = get_pointer(hOHTarget);
00089             (*pTarget.*(M))(cast<AH>(ohValue));
00090             }
00091 
00092 
00093     // ----- Object interface -----------------------------------------------
00094 
00095     public:
00096         /**
00097         * {@inheritDoc}
00098         */
00099         virtual bool equals(Object::View v) const
00100             {
00101             return instanceof<typename TypedUpdater<A, C, M, AH, OH>::View>(v);
00102             }
00103 
00104         /**
00105         * {@inheritDoc}
00106         */
00107         virtual size32_t hashCode() const
00108             {
00109             return Class::getClassName(this)->hashCode();
00110             }
00111 
00112         /**
00113         * {@inheritDoc}
00114         */
00115         virtual TypedHandle<const String> toString() const
00116             {
00117             return Class::getClassName(this);
00118             }
00119     };
