00029     : public interface_spec<MemberListener,

00032     // ----- MemberListener interface ---------------------------------------
00033 
00034     public:
00035         /**
00036         * Invoked when a Member has joined the service.
00037         *
00038         * Note: this event could be called during the service restart on the
00039         * local node ({@link MemberEvent#isLocal() evt.isLocal()}) in which
00040         * case the listener's code should not attempt to use any clustered
00041         * cache or service functionality.
00042         *
00043         * The most critical situation arises when a number of threads are
00044         * waiting for a local service restart, being blocked by a Service
00045         * object synchronization monitor. Since the Joined event should be
00046         * fired only once, it is called on an event dispatcher thread
00047         * <b>while holding a synchronization monitor</b>. An attempt to use
00048         * other clustered service functionality during this local event
00049         * notification may result in a deadlock.
00050         *
00051         * @param vEvt  the MemberEvent::member_joined event
00052         */
00053         virtual void memberJoined(MemberEvent::View vEvt) = 0;
00054 
00055         /**
00056         * Invoked when a Member is leaving the service.
00057         *
00058         * @param vEvt  the MemberEvent::member_leaving event
00059         */
00060         virtual void memberLeaving(MemberEvent::View vEvt) = 0;
00061 
00062         /**
00063         * Invoked when a Member has left the service.
00064         *
00065         * Note: this event could be called during the service restart on the
00066         * local node ({@link MemberEvent#isLocal() evt.isLocal()}) in which
00067         * case the listener's code should not attempt to use any clustered
00068         * cache or service functionality.
00069         *
00070         * @param vEvt  the MemberEvent::member_joined event
00071         */
00072         virtual void memberLeft(MemberEvent::View vEvt) = 0;
00073     };
