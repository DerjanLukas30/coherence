00023 /**
00024 * Reference counting "smart pointer" class for use with Coherence managed
00025 * objects. Managed objects manage their own life cycle, rather the relying on
00026 * user code to identify when they should be deleted.  The TypedHandle class
00027 * can be used with any class implementing the coherence::lang::Object
00028 * interface.
00029 *
00030 * By convention each Object implementation will include two typedefs for
00031 * TypedHandle to make their usage easier. For instance the class "Foo" would
00032 * define Foo::Handle as a TypedHandle<Foo>. The second typedef is called a
00033 * View, and is analogous to a pointer to a const, that is Foo::View is
00034 * TypedHandle<const Foo>.
00035 *
00036 * TypedHandles are not thread-safe, and should not be used in a multi-threaded
00037 * context even if protected by external synchronization. They should only be
00038 * used as local variables. The most common place where thread-safe handles are
00039 * needed is for object data members. For these cases the Coherence API
00040 * includes thread-safe handles/views, namely MemberHandle/View and
00041 * WeakHandle/View.
00042 *
00043 * @see Object
00044 * @see MemberHandle
00045 * @see MemberView
00046 * @see WeakHandle
00047 * @see WeakView
00048 * @see FinalHandle
00049 * @see FinalView
00050 *
00051 * @author mf 2007.07.05
00052 */

00561 /**
00562 * Perform a dynamic cast the pointer associated with the TypedHandle
00563 * to a the specified handle/view type.
00564 *
00565 * @param th      the TypedHandle from which to perform the cast
00566 * @param fThrow  true if an exception is to be thrown on a failed cast
00567 *
00568 * @return the casted pointer, or NULL if the cast fails and fThrow is false
00569 *
00570 * @throws ClassCastException if the cast fails and fThrow of is true
00571 */
00572 template<class D, class T>
00573 D cast(const TypedHandle<T>& th, bool fThrow = true)
00574     {
00575     if (NULL == th)
00576         {
00577         return D(); // NULL
00578         }
00579 
00580     typedef typename D::ValueType DT;
00581     typedef typename DT::this_spec DTS;
00582     DT* pD;
00583     if (assignment<DT, T>::allowed)
00584         {
00585         // free compile time cast
00586         pD = assignment<DT, T>::safe(get_pointer(th));
00587         }
00588     else // runtime cast
00589         {
00590         pD = (DT*) th->_cast(COH_CLASS_ID(DTS));
00591         }
00592 
00593     if (NULL == pD)
00594         {
00595         if (fThrow)
00596             {
00597             coh_throw_class_cast(typeid(DT), typeid(*th));
00598             }
00599         return D(); // NULL
00600         }
00601 
00602     if (constness<T>::applied)
00603         {
00604         // from View to View or Holder(View)
00605         return TypedHandle<DT>(pD, th);
00606         }
00607 
00608     // from Handle to Handle, View, or Holder(Handle)
00609     TypedHandle<typename DT::Handle::ValueType> hD(
00610             const_cast<typename DT::Handle::ValueType*>(pD), th);
00611     return hD;
00612     }
