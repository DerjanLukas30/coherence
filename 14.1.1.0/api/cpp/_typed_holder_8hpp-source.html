00022 /**
00023 * A handle implementation which supports referencing Objects as either
00024 * Handles or Views. TypedHolder can safely be used in place of View
00025 * but adds the ability to attempt a safe down cast to a Handle. This differs
00026 * from a const_cast in that the down cast will only succeed if the
00027 * TypedHolder had been assigned from a Handle, an assignment from a View
00028 * results in a TypedHolder whose down cast operation will fail with a
00029 * ClassCastException.
00030 *
00031 * TypedHolder is not for general use, instead it is a convenience handle type
00032 * which can be used for container like classes, which need to be able to
00033 * contain both Handles and Views.
00034 *
00035 * As with TypedHandles, TypedHolders are not thread-safe, and should not be
00036 * used in a multi-threaded context even if protected by external
00037 * synchronization. They should only be used as local variables. The most
00038 * common place where thread-safe holders are needed is for object data
00039 * members. For these cases the Coherence API includes a thread-safe variant
00040 * MemberHolder.
00041 *
00042 * @author mf  2008.01.09
00043 *
00044 * @see MemberHolder
00045 * @see WeakHolder
00046 * @see FinalHolder
00047 */

00281         * Perform the specified action.
00282         *
00283         * @param nAction  the action to perform
00284         */
00285         void performAction(Action nAction)
00286             {
00287             switch (nAction)
00288                 {
00289                 case action_flip:
00290                     m_cpo->_attach(/*fEscaped*/ false);
00291                     // fall through
00292                 case action_detach:
00293                     if (m_fView)
00294                         {
00295                         m_cpo->_detach(/*fEscaped*/ false);
00296                         }
00297                     else
00298                         {
00299                         const_cast<T*>(m_cpo)->_detach(/*fEscaped*/ false);
00300                         }
00301                     // fall through
00302                 case action_none:
00303                 default:
00304                     break;
00305                 }
00306             }

00423 /**
00424 * Perform a dynamic cast the pointer associated with the TypedHolder
00425 * to a the specified handle/view type.
00426 *
00427 * @param th      the TypedHolder from which to perform the cast
00428 * @param fThrow  true if an exception is to be thrown on a failed cast
00429 *
00430 * @return the casted pointer, or NULL if the cast fails and fThrow is false
00431 *
00432 * @throws ClassCastException if the cast fails and fThrow is true
00433 */
00434 template<class D, class T>
00435 D cast(const TypedHolder<T>& th, bool fThrow = true)
00436     {
00437     TypedHandle<T> h = th.getHandle();
00438     if (NULL == h)
00439         {
00440         TypedHandle<const T> v = th;
00441 
00442         if (NULL == v)
00443             {
00444             // from NULL to any type is allowed
00445             return D(); // NULL
00446             }
00447 
00448         // input is a non-NULL view
00449         if (constness<typename D::ValueType>::applied)
00450             {
00451             // input type is a holder(view), output type is a view or holder
00452             // return a view and let it be converted if necessary to a holder
00453             // producing a holder(handle) can't happen from here, but we've
00454             // already proved that the input type is a view (NULL == h)
00455             TypedHandle<typename D::ValueType> vD(
00456                 (typename D::ValueType*) get_pointer(
00457                         cast<typename D::ValueType::View>(v, fThrow)), v);
00458             return vD;
00459             }
00460         else
00461             {
00462             // output type is a handle; input type is a view (error)
00463             if (fThrow)
00464                 {
00465                 coh_throw_const_cast(typeid(typename D::ValueType), typeid(*th));
00466                 }
00467             return D(); // NULL
00468             }
00469         }
00470 
00471     // input type is a holder(handle), output type can be handle/view/holder;
00472     // cast from on the handle, and allow conversion to produce expected
00473     // result type
00474     return cast<typename D::ValueType::Handle>(h, fThrow);
00475     }
