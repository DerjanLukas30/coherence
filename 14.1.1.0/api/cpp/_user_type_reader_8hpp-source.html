00023 /**
00024 * The UserTypeReader implementation is a contextually-aware PofReader whose
00025 * purpose is to advance through the properties of a value of a specified user
00026 * type. The "contextual awareness" refers to the fact that the UserTypeReader
00027 * maintains state about the type identifier and version of the user type, the
00028 * parser's property index position within the user type value, and a
00029 * PofContext that may differ from the PofContext that provided the
00030 * PofSerializer which is using this UserTypeReader to parse a user type.
00031 *
00032 * @author jh  2008.04.11
00033 */
00034 class COH_EXPORT UserTypeReader
00035     : public class_spec<UserTypeReader,
00036         extends<PofBufferReader> >
00037     {

00046         * @param hIn         the BufferInput that contains the user type
00047         *                    data, except for the user type id itself (which
00048         *                    is passed passed as a constructor argument)
00049         * @param vCtx        the PofContext to use for parsing the user type
00050         *                    property values within the user type that this
00051         *                    parser will be parsing
00052         * @param nTypeId     the type id of the user type
00053         * @param nVersionId  the version id of the user type
00054         *
00055         * @throws IOException  if an I/O error occurs
00056         */
00057         UserTypeReader(ReadBuffer::BufferInput::Handle hIn,
00058                 PofContext::View vCtx, int32_t nTypeId, int32_t nVersionId);
